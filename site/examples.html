Sample MCP tool call

Request

{
  "tool": "queryFeatures",
  "arguments": {
    "environment": "enterprise",
    "layer": "Park Facilities",
    "where": "STATUS='Active'",
    "outFields": ["FACILITYID","NAME","STATUS"]
  }
}

Response

{
  "type": "FeatureCollection",
  "features": [
    {
      "type": "Feature",
      "properties": {
        "FACILITYID": "REC-0142",
        "NAME": "Golden Gate Park Clubhouse",
        "STATUS": "Active"
      },
      "geometry": { "type": "Point", "coordinates": [-122.47, 37.77] }
    }
  ]
}
